Ejercicio 22

22.-Cree una función que capitalice la última letra de cada palabra.
capLast("hello") ➞ "hellO" 


	function capLast(string) {
		let arr = string.split(' ');
		let farr = arr.map((item) => {
			let x = item.split('');
			let len = x.length - 1;
			x[len] = x[len].toUpperCase();
			x = x.join('');
			return x;
		});
		return farr.join(' ');
	}
	console.log(capLast('hello hello'));
